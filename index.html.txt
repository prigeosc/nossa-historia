const inicio=new Date("2024-10-25");

function atualizar(){

const agora=new Date();
const diff=agora-inicio;

document.getElementById('dias').innerHTML=Math.floor(diff/86400000);
document.getElementById('horas').innerHTML=Math.floor(diff/3600000);
document.getElementById('minutos').innerHTML=Math.floor(diff/60000);
document.getElementById('segundos').innerHTML=Math.floor(diff/1000);

}

setInterval(atualizar,1000);
atualizar();
